    listen 8080;
    server_name seonghyeon;
    root ./www;
    error_page ./www/errors/404.html;
    methods GET POST DELETE;

    location / {
        root ./www;
        autoindex on;
        methods GET;
    }

    location /upload {
        root ./www/uploads;
        autoindex off;
        methods POST DELETE;
    }
}
